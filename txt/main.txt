    elif opcao == "3":
        if renda == 0:
            print("⚠️ Informe sua renda primeiro (opção 1).")
        else:
            valor = float(input("Valor do produto: "))
            simular_compra(renda, valor)

    elif opcao == "0":
        print("Encerrando...")
        break

    else:
        print("Opção inválida.")

import customtkinter as ctk
from orcamento import calcular_orcamento
from cartao import analisar_cartao
from simulador import simular_compra

app = ctk.CTk()
app.title("Assistente Financeiro")
app.geometry("400x500")


renda = 0
limite = 0

# limpar tela
def limpar_tela():
    for widget in app.winfo_children():
        widget.destroy()

# MENU
def mostrar_menu():
    limpar_tela()

    titulo = ctk.CTkLabel(app, text="Assistente Financeiro", font=("Arial", 20))
    titulo.pack(pady=20)

    ctk.CTkButton(app, text="Orçamento", command=tela_orcamento).pack(pady=10)
    ctk.CTkButton(app, text="Cartão de Crédito", command=tela_cartao).pack(pady=10)
    ctk.CTkButton(app, text="Simular Compra", command=tela_simulador).pack(pady=10)

# TELA ORÇAMENTO
def tela_orcamento():
    limpar_tela()

    ctk.CTkLabel(app, text="Digite sua renda").pack(pady=10)
    entry = ctk.CTkEntry(app)
    entry.pack(pady=10)

    resultado = ctk.CTkLabel(app, text="")
    resultado.pack(pady=10)

    def calcular():
        global renda
        try:
            renda = float(entry.get())
            dados = calcular_orcamento(renda)

            resultado.configure(
                text=f"""
Essenciais: R$ {dados['essencial']:.2f}
Lazer: R$ {dados['lazer']:.2f}
Investimentos: R$ {dados['investimento']:.2f}

{dados['diagnostico']}
"""
            )
        except:
            resultado.configure(text="Digite um valor válido")

    ctk.CTkButton(app, text="Calcular", command=calcular).pack(pady=10)
    ctk.CTkButton(app, text="Voltar", command=mostrar_menu).pack(pady=10)

# TELA CARTÃO
def tela_cartao():
    limpar_tela()

    ctk.CTkLabel(app, text="Limite do cartão").pack(pady=10)
    entry = ctk.CTkEntry(app)
    entry.pack(pady=10)

    resultado = ctk.CTkLabel(app, text="")
    resultado.pack(pady=10)

    def analisar():
        global limite
        try:
            limite = float(entry.get())


            if renda == 0:
                resultado.configure(text="Informe a renda primeiro")
                return

            dados = analisar_cartao(renda, limite)

            resultado.configure(
                text=f"""
Limite: {dados['porcentagem']:.1f}% da renda

{dados['status']}
{dados['dica']}
"""
            )
        except:
            resultado.configure(text="Digite um valor válido")

    ctk.CTkButton(app, text="Analisar", command=analisar).pack(pady=10)
    ctk.CTkButton(app, text="Voltar", command=mostrar_menu).pack(pady=10)

# TELA SIMULADOR
def tela_simulador():
    limpar_tela()

    ctk.CTkLabel(app, text="Valor do produto").pack(pady=10)
    entry = ctk.CTkEntry(app)
    entry.pack(pady=10)

    resultado = ctk.CTkLabel(app, text="")
    resultado.pack(pady=10)

    def simular():
        try:
            valor = float(entry.get())

            if renda == 0:
                resultado.configure(text="Informe a renda primeiro")
                return

            dados = simular_compra(renda, valor)

            resultado.configure(
                text=f"""
Parcelas: {dados['parcelas']}x
Valor: R$ {dados['valor_parcela']:.2f}

{dados['status']}
"""
            )
        except:
            resultado.configure(text="Digite um valor válido")

    ctk.CTkButton(app, text="Simular", command=simular).pack(pady=10)
    ctk.CTkButton(app, text="Voltar", command=mostrar_menu).pack(pady=10)

# iniciar
mostrar_menu()
app.mainloop()
